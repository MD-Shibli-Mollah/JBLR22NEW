    SUBROUTINE JBL.DR.CR.NARRATIVE.TEXT
    $INSERT GLOBUS.BP I_COMMON
    $INSERT GLOBUS.BP I_EQUATE
    $INSERT GLOBUS.BP I_ENQUIRY.COMMON

    Y.TEXT = O.DATA

    LOOP
    WHILE LEN(Y.TEXT) <> 0
        Y.TEMP.DATA = FIELD(Y.TEXT, " ", 1, 20)
        Y.DATA<-1> = FIELD(Y.TEXT, " ", 1, 20)
        Y.TEXT = Y.TEXT[LEN(Y.TEMP.DATA)+1,LEN(Y.TEXT)]
        Y.TEXT = TRIM(Y.TEXT)
    REPEAT
    Y.DATA = TRIM(Y.DATA)
    CONVERT FM TO "*" IN Y.DATA
    O.DATA = Y.DATA
    RETURN

END
